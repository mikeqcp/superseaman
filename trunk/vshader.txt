#version 130
#define GLM_SWIZZLE GLM_SWIZZLE_FULL

uniform mat4 P;
uniform mat4 V;
uniform mat4 M;
uniform vec4 lightPos;

uniform vec4 ambient;
uniform vec4 diffuse;
uniform vec4 specular;

uniform vec4 clipPlane;

in vec4 vertex;
in vec4 normal;

out vec4 inormal;
out vec4 ilight;
out vec4 iv;


void main(void){

	vec4 n = normalize(V*M*normal);
	vec4 l = normalize(V*M*vertex - V*lightPos);
	
	gl_Position = P*V*M*vertex;
	//gl_ClipDistance[0] = dot(clipPlane, vertex);

	if(vertex.y < clipPlane.y) gl_ClipVertex = vertex;
	else gl_ClipVertex = vec4(0,0,0,0);

	inormal = n;
	ilight = l;
	iv = V*vec4(0, 5.0f, 7.0f, 1);

}
