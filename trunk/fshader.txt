#version 130

uniform vec4 ambient;
uniform vec4 diffuse;
uniform vec4 specular;

out vec4 pixelColor;
in vec4 inormal;
in vec4 ilight;


void main(void) {

	vec4 n = normalize(inormal);
	vec4 l = normalize(ilight);
	
	float nl = dot(n, l);
	nl  = max(nl, 0);

	pixelColor = ambient + diffuse*nl;

}
